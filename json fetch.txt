import React, { useEffect, useState } from "react";

function StudentListjson() {
  const [students, setStudents] = useState([]);

  useEffect(() => {
    fetch("/students.json") // file should be in public/ folder
      .then((res) => res.json())
      .then((data) => setStudents(data));
  }, []);

  return (
    <div>
      <h2>Student Records</h2>
      <ul>
        {students.map((student) => (
          <li key={student.id}>
            {student.name} â€” Roll No: {student.rollNo}, Dept: {student.department}
          </li>
        ))}
      </ul>
    </div>
  );
}

export default StudentListjson;
