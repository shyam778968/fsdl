Form.jsx------------------------------------------------------------
import React, { useState } from "react";
import "./Form.css";
import StudentList from "./StudentList"; 

function Form() {
  // State variables to store input values
  const [name, setName] = useState("");
  const [rollNo, setRollNo] = useState("");
  const [department, setDepartment] = useState("");

  // State to store the list of all students
  const [students, setStudents] = useState([]);

  // Function runs when the form is submitted
  const handleSubmit = (e) => {
    e.preventDefault(); // Prevent page reload

    // Create a new student record object
    const newStudent = {
      name,
      rollNo,
      department,
    };

    // Add the new student to the existing list
    setStudents([...students, newStudent]);

    // Clear all input fields after submission
    setName("");
    setRollNo("");
    setDepartment("");
  };

  return (
    <div className="form-container">
      <h2>Add New Student</h2>

      {/* Student input form */}
      <form onSubmit={handleSubmit} className="student-form">
        <div className="form-group">
          <label>Name:</label>
          <input
            type="text"
            value={name}
            placeholder="Enter student name"
            onChange={(e) => setName(e.target.value)}
            required
          />
        </div>

        <div className="form-group">
          <label>Roll No:</label>
          <input
            type="text"
            value={rollNo}
            placeholder="Enter roll number"
            onChange={(e) => setRollNo(e.target.value)}
            required
          />
        </div>

        <div className="form-group">
          <label>Department:</label>
          <input
            type="text"
            value={department}
            placeholder="Enter department"
            onChange={(e) => setDepartment(e.target.value)}
            required
          />
        </div>

        <button type="submit" className="btn-submit">
          Add Student
        </button>
      </form>

      <h3>Student Records</h3>

      {/* Pass the list of students as a prop to StudentList */}
      <StudentList students={students} />
    </div>
  );
}

export default Form;

import React from "react";

function StudentList({ students }) {
  // If there are no students
  if (students.length === 0) {
    return (
      <ul className="student-list">
        <li>No students added yet.</li>
      </ul>
    );
  } 
  // Otherwise (if students exist)
  else {
    return (
      <ul className="student-list">
        {students.map((student, index) => (
          <li key={index}>
            <strong>{student.name}</strong> â€” Roll No: {student.rollNo}, Dept: {student.department}
          </li>
        ))}
      </ul>
    );
  }
}

export default StudentList;
